  <script>

    // ------------------------

    // 1) Hodiny & dátum

    // ------------------------

    function updateClock() {

      const t = new Date();

      const hh = t.getHours().toString().padStart(2,'0');

      const mm = t.getMinutes().toString().padStart(2,'0');

      const ss = t.getSeconds().toString().padStart(2,'0');

      document.getElementById('time').textContent = `${hh}:${mm}`;

      document.getElementById('seconds').textContent = ss;


      const days = ["nedeľa","pondelok","utorok","streda","štvrtok","piatok","sobota"];

      const dayName = days[t.getDay()];

      const dd = t.getDate().toString().padStart(2,'0');

      const mn = (t.getMonth()+1).toString().padStart(2,'0');

      const yyyy = t.getFullYear();

      document.getElementById('date').textContent = `${dayName}, ${dd}.${mn}.${yyyy}`;

    }

    setInterval(updateClock, 1000);

    updateClock();


    // ------------------------

    // 2) Prehrávanie streamu

    // ------------------------

    const player = document.getElementById('player');

    const btn = document.getElementById('play-btn');


    // Po úspešnom štarte prehrávania schovať tlačidlo

    player.addEventListener('playing', () => {

      btn.style.display = 'none';

    });


    // Ak prehrávanie zlyhá, ukázať chybu v tlačidle

    player.addEventListener('error', () => {

      btn.textContent = 'Chyba pri prehrávaní rádia';

      btn.style.display = 'block';

    });


    // Keď sa tlačidlo stlačí, spusti prehrávanie

    btn.addEventListener('click', () => {

      btn.disabled = true;

      btn.textContent = 'Pripájam...';

      player.play().catch(_=>{

        btn.textContent = 'Žiadne audio zdroj';

        btn.disabled = false;

      });

    });


    // Autoplay (pokúsi sa bez tlačidla)

    window.addEventListener('load', () => {

      player.play()

        .catch(_=>{

          // ak autoplay prehliadač zablokuje, ukážeme tlačidlo

          btn.style.display = 'block';

        });

    });

  </script>
